# za38-cli 实现计划

> **给执行者：** 使用 compose:subagent（推荐）或 compose:execute 按任务逐个实现。步骤使用复选框（`- [ ]`）跟踪进度。

**目标：** 构建一个企业级编码智能体 CLI，opentui（TS/Node）表现层 + deepagents（Python）agent 内核，通过 JSON-RPC over stdio 通信，作为单个 npm 包分发。MVP 聚焦核心编码 agent 功能（参照 dcode），za38 垂域命令后续整合。

**架构：** 薄协议 + 智能内核。Node/Bun 将 Python 作为子进程启动。IPC 是换行分隔的 JSON-RPC 2.0，走 stdin/stdout。Python 拥有全部 agent 逻辑（deepagents + LangGraph），内置 filesystem/shell/todo 等编码工具。Node 通过 opentui + React 渲染 TUI。

**技术栈：** TypeScript（Bun 为主，兼容 Node 22+）、@opentui/core + @opentui/react（vendor 引入）、React 19、Python 3.12、deepagents、LangChain、LangGraph、uv（Python 包管理器）

## 全局约束

- Python >=3.11（deepagents 要求），通过 uv 目标 3.12
- 运行时：Bun >=1.3.0（主力，opentui 原生 FFI 路径）；Node 22+ 加 `--experimental-ffi` 也可用（opentui 有双后端 FFI 抽象层）
- @opentui/core 已发布到 npm（^0.4.3）；@opentui/react 是私有包 —— 必须 vendor 引入
- stdout 仅供 JSON-RPC 协议消息使用；Python 调试输出走 stderr
- 所有 JSON-RPC 消息为换行分隔的 UTF-8 JSON，每行一个对象
- deepagents >=0.6.12，langchain >=1.3.12
- npm 包名：`@za38/cli`，二进制名：`za38`
- Python 包名：`za38-agent`，导入名：`za38_agent`
- 测试框架：pytest（Python）、bun test（Node/TS）
- **MVP 范围**：核心编码 agent 功能（filesystem/shell/todo 工具），za38 垂域命令（codegen/scaffold/ui/infra）后续整合

```python
# packages/agent/za38_agent/server.py
"""JSON-RPC 2.0 server，通过 stdin/stdout 通信（换行分隔）。"""
import asyncio
import json
import sys
import uuid
from typing import Any


class JsonRpcServer:
    """异步 JSON-RPC server，从 stdin 读取，写入 stdout。"""

    def __init__(self) -> None:
        self.agent = None  # 在任务 3 中设置
        self._running = True
        self._handlers = {
            "initialize": self._handle_initialize,
            "query": self._handle_query,
            "cancel": self._handle_cancel,
            "respond": self._handle_respond,
            "shutdown": self._handle_shutdown,
        }

    async def run(self) -> None:
        """主循环：从 stdin 读取行，分发消息。"""
        reader = asyncio.StreamReader()
        protocol = asyncio.StreamReaderProtocol(reader)
        loop = asyncio.get_event_loop()
        await loop.connect_read_pipe(lambda: protocol, sys.stdin)

        while self._running:
            line = await reader.readline()
            if not line:
                break
            try:
                msg = json.loads(line.decode("utf-8"))
                await self.dispatch(msg)
            except json.JSONDecodeError:
                await self.send_error(None, -32700, "Parse error")
            except Exception as e:
                msg_id = msg.get("id") if "msg" in dir() else None
                await self.send_error(msg_id, -32603, str(e))

    async def dispatch(self, msg: dict[str, Any]) -> None:
        """将 JSON-RPC 消息路由到对应的 handler。"""
        method = msg.get("method")
        msg_id = msg.get("id")
        params = msg.get("params", {})

        handler = self._handlers.get(method)
        if handler is None:
            await self.send_error(msg_id, -32601, f"Method not found: {method}")
            return

        params_with_id = {**params, "_id": msg_id}
        result = await handler(params_with_id)
        if result is not None and msg_id is not None:
            await self.send_response(msg_id, result)

    async def send(self, msg: dict[str, Any]) -> None:
        """将 JSON-RPC 消息写入 stdout。"""
        data = (json.dumps(msg, ensure_ascii=False) + "\n").encode("utf-8")
        sys.stdout.buffer.write(data)
        sys.stdout.buffer.flush()

    async def send_response(self, msg_id: int | None, result: Any) -> None:
        await self.send({"jsonrpc": "2.0", "result": result, "id": msg_id})

    async def send_error(self, msg_id: int | None, code: int, message: str) -> None:
        await self.send({"jsonrpc": "2.0", "error": {"code": code, "message": message}, "id": msg_id})

    async def send_notification(self, method: str, params: dict[str, Any]) -> None:
        await self.send({"jsonrpc": "2.0", "method": method, "params": params})

    async def _handle_initialize(self, params: dict) -> dict:
        return {
            "server_info": {"name": "za38-agent", "version": "0.1.0"},
            "capabilities": {"streaming": True, "hitl": True},
        }

    async def _handle_query(self, params: dict) -> dict | None:
        thread_id = params.get("thread_id") or str(uuid.uuid4())
        msg_id = params.get("_id")

        # 立即发送响应（请求已接受）
        if msg_id is not None:
            await self.send_response(msg_id, {"thread_id": thread_id, "accepted": True})

        # 如果 agent 已设置，流式发送事件（任务 3 接入）
        if self.agent is not None:
            await self._stream_agent_response(params["message"], thread_id)
        else:
            # Echo 模式：只发送 done
            await self.send_notification("stream/done", {
                "thread_id": thread_id,
                "usage": {"input_tokens": 0, "output_tokens": 0},
            })

        return None  # 响应已发送

    async def _stream_agent_response(self, message: str, thread_id: str) -> None:
        """流式发送 agent 响应 —— 任务 3 中实现。"""
        raise NotImplementedError("Agent streaming not yet implemented")

    async def _handle_cancel(self, params: dict) -> dict:
        return {"cancelled": True}

    async def _handle_respond(self, params: dict) -> dict:
        return {"accepted": True}

    async def _handle_shutdown(self, params: dict) -> dict:
        self._running = False
        return {}
```

```typescript
// packages/cli/src/ipc/client.ts
import { EventEmitter } from "node:events"
import type { Readable, Writable } from "node:stream"
import type { JsonRpcMessage } from "@za38/protocol"

/**
 * JSON-RPC 2.0 客户端，通过 stdin/stdout 通信（换行分隔）。
 * 向 Python 进程的 stdin 写请求，从 stdout 读响应/通知。
 */
export class IpcClient extends EventEmitter {
  private nextId = 1
  private pending = new Map<number, { resolve: (v: unknown) => void; reject: (e: Error) => void }>()
  private buffer = ""

  constructor(
    private stdin: Writable,
    private stdout: Readable,
  ) {
    super()
    this.stdout.on("data", (chunk: Buffer) => this.onData(chunk))
    this.stdout.on("end", () => this.emit("close"))
    this.stdout.on("error", (err) => this.emit("error", err))
  }

  private onData(chunk: Buffer): void {
    this.buffer += chunk.toString("utf-8")
    const lines = this.buffer.split("\n")
    this.buffer = lines.pop() ?? ""
    for (const line of lines) {
      if (!line.trim()) continue
      try {
        const msg = JSON.parse(line) as JsonRpcMessage
        this.handleMessage(msg)
      } catch (err) {
        this.emit("error", new Error(`JSON-RPC 消息解析失败: ${line}`))
      }
    }
  }

  private handleMessage(msg: JsonRpcMessage): void {
    if ("method" in msg && msg.method) {
      // 通知（无 id）—— Python 只发通知给我们
      this.emit(msg.method, msg.params)
    } else if ("id" in msg && msg.id !== undefined) {
      // 响应
      const pending = this.pending.get(msg.id)
      if (pending) {
        this.pending.delete(msg.id)
        if ("error" in msg && msg.error) {
          pending.reject(new Error(msg.error.message))
        } else {
          pending.resolve(msg.result)
        }
      }
    }
  }

  /**
   * 发送 JSON-RPC 请求并等待响应。
   */
  async call(method: string, params: Record<string, unknown> = {}): Promise<unknown> {
    const id = this.nextId++
    return new Promise((resolve, reject) => {
      this.pending.set(id, { resolve, reject })
      this.send({ jsonrpc: "2.0", method, params, id })
    })
  }

  /**
   * 向 agent 内核发送 query。
   */
  async query(message: string, threadId?: string): Promise<{ thread_id: string; accepted: boolean }> {
    return this.call("query", { message, thread_id: threadId }) as Promise<{ thread_id: string; accepted: boolean }>
  }

  /**
   * 发送通知（不需要响应）。
   */
  notify(method: string, params: Record<string, unknown> = {}): void {
    this.send({ jsonrpc: "2.0", method, params })
  }

  private send(msg: Record<string, unknown>): void {
    this.stdin.write(JSON.stringify(msg) + "\n")
  }

  /**
   * 关闭时清理待处理请求。
   */
  destroy(): void {
    for (const [, { reject }] of this.pending) {
      reject(new Error("Connection closed"))
    }
    this.pending.clear()
    this.removeAllListeners()
  }
}
```

# 在 _stream_agent_response 方法中替换 NotImplementedError：

    async def _stream_agent_response(self, message: str, thread_id: str) -> None:
        """调用 graph.astream 并翻译事件，流式发送 agent 响应。"""
        from langchain_core.messages import HumanMessage

        async for event in self.agent.astream(
            {"messages": [HumanMessage(content=message)]},
            config={"configurable": {"thread_id": thread_id}},
            stream_mode=["messages", "updates"],
        ):
            notification = self._translate_event(event, thread_id)
            if notification:
                method, params = notification
                await self.send_notification(method, params)

        await self.send_notification("stream/done", {
            "thread_id": thread_id,
            "usage": {"input_tokens": 0, "output_tokens": 0},
        })

    def _translate_event(self, event: tuple, thread_id: str) -> tuple[str, dict] | None:
        """将 LangGraph 流式事件翻译为 JSON-RPC 通知。

        LangGraph 事件是 (namespace, stream_mode, data) 三元组。
        """
        namespace, stream_mode, data = event

        if stream_mode == "messages":
            chunk, metadata = data
            chunk_type = type(chunk).__name__

            if chunk_type == "AIMessageChunk":
                # 文本内容
                if chunk.content:
                    return ("stream/text", {"text": chunk.content, "thread_id": thread_id})
                # 工具调用片段
                if hasattr(chunk, "tool_call_chunks") and chunk.tool_call_chunks:
                    tc = chunk.tool_call_chunks[0]
                    if tc.get("name"):
                        return ("stream/tool_start", {
                            "tool_name": tc["name"],
                            "tool_id": tc.get("id", ""),
                            "args": {},
                            "thread_id": thread_id,
                        })
                    elif tc.get("args"):
                        return ("stream/tool_chunk", {
                            "tool_id": tc.get("id", ""),
                            "chunk": tc["args"],
                            "thread_id": thread_id,
                        })

            elif chunk_type == "ToolMessage":
                return ("stream/tool_result", {
                    "tool_id": getattr(chunk, "tool_call_id", ""),
                    "result": chunk.content if isinstance(chunk.content, str) else str(chunk.content),
                    "thread_id": thread_id,
                })

        elif stream_mode == "updates":
            if isinstance(data, dict) and "__interrupt__" in data:
                return ("stream/interrupt", {
                    "thread_id": thread_id,
                    "description": "工具执行需要审批",
                })

        return None

// packages/cli/src/tui/hooks/messageReducer.ts

export type MessageRole = "user" | "ai" | "tool" | "error"

export interface BaseMessage {
  id: string
  role: MessageRole
}

export interface UserMessage extends BaseMessage {
  role: "user"
  content: string
}

export interface AIMessage extends BaseMessage {
  role: "ai"
  content: string
  streaming: boolean
}

export interface ToolMessage extends BaseMessage {
  role: "tool"
  name: string
  args: Record<string, unknown>
  result?: string
  status: "running" | "done" | "error"
}

export interface ErrorMessage extends BaseMessage {
  role: "error"
  content: string
}

export type Message = UserMessage | AIMessage | ToolMessage | ErrorMessage

export type MessageAction =
  | { type: "USER_MESSAGE"; text: string }
  | { type: "APPEND_TEXT"; text: string }
  | { type: "TOOL_START"; tool_name: string; tool_id: string; args: Record<string, unknown> }
  | { type: "TOOL_RESULT"; tool_id: string; result: string; error?: string }
  | { type: "PLAN_UPDATE"; todos: { content: string; status: string }[] }
  | { type: "DONE" }
  | { type: "ERROR"; message: string }

let _id = 0
const nextId = () => `msg-${++_id}`

export function messageReducer(state: Message[], action: MessageAction): Message[] {
  switch (action.type) {
    case "USER_MESSAGE":
      return [...state, { id: nextId(), role: "user", content: action.text } as UserMessage]

    case "APPEND_TEXT": {
      const last = state[state.length - 1]
      if (last && last.role === "ai" && last.streaming) {
        return [...state.slice(0, -1), { ...last, content: last.content + action.text }]
      }
      return [...state, { id: nextId(), role: "ai", content: action.text, streaming: true } as AIMessage]
    }

    case "TOOL_START":
      return [...state, {
        id: action.tool_id,
        role: "tool",
        name: action.tool_name,
        args: action.args,
        status: "running",
      } as ToolMessage]

    case "TOOL_RESULT":
      return state.map((m) =>
        m.id === action.tool_id && m.role === "tool"
          ? { ...m, result: action.result, status: action.error ? "error" : "done" }
          : m
      )

    case "PLAN_UPDATE":
      // TODO: 在面板中渲染计划 —— 暂时不操作
      return state

    case "DONE": {
      const last = state[state.length - 1]
      if (last && last.role === "ai" && last.streaming) {
        return [...state.slice(0, -1), { ...last, streaming: false }]
      }
      return state
    }

    case "ERROR":
      return [...state, { id: nextId(), role: "error", content: action.message } as ErrorMessage]

    default:
      return state
  }
}

// packages/cli/src/tui/hooks/useAgentStream.ts
import { useReducer, useState, useEffect, useCallback } from "react"
import type { IpcClient } from "../../ipc/client"
import { messageReducer, type Message } from "./messageReducer"
import { Notification } from "@za38/protocol"

export type AgentStatus = "idle" | "thinking" | "error"

export function useAgentStream(client: IpcClient) {
  const [messages, dispatch] = useReducer(messageReducer, [] as Message[])
  const [status, setStatus] = useState<AgentStatus>("idle")
  const [usage, setUsage] = useState({ input_tokens: 0, output_tokens: 0 })

  useEffect(() => {
    const handlers: Record<string, (params: any) => void> = {
      [Notification.STREAM_TEXT]: (p) => dispatch({ type: "APPEND_TEXT", text: p.text }),
      [Notification.STREAM_TOOL_START]: (p) => dispatch({ type: "TOOL_START", tool_name: p.tool_name, tool_id: p.tool_id, args: p.args }),
      [Notification.STREAM_TOOL_RESULT]: (p) => dispatch({ type: "TOOL_RESULT", tool_id: p.tool_id, result: p.result, error: p.error }),
      [Notification.STREAM_PLAN]: (p) => dispatch({ type: "PLAN_UPDATE", todos: p.todos }),
      [Notification.STREAM_DONE]: (p) => {
        setStatus("idle")
        setUsage(p.usage)
        dispatch({ type: "DONE" })
      },
      [Notification.STREAM_ERROR]: (p) => {
        setStatus("error")
        dispatch({ type: "ERROR", message: p.message })
      },
    }

    for (const [method, handler] of Object.entries(handlers)) {
      client.on(method, handler)
    }

    return () => {
      for (const [method, handler] of Object.entries(handlers)) {
        client.off(method, handler)
      }
    }
  }, [client])

  const sendMessage = useCallback((text: string) => {
    if (!text.trim()) return
    setStatus("thinking")
    dispatch({ type: "USER_MESSAGE", text })
    client.query(text)
  }, [client])

  return { messages, status, usage, sendMessage }
}

MVP 版本：启用 deepagents 内置的编码工具（filesystem + shell），
参照 dcode 的核心编码能力。za38 垂域命令后续整合。

from deepagents import create_deep_agent, FilesystemMiddleware
from deepagents.backends import LocalShellBackend
from langchain_core.language_models import BaseChatModel

_PROMPT_PATH = Path(__file__).parent / "prompts" / "system_prompt.md"


def _load_system_prompt() -> str:
    """从打包的 markdown 文件加载系统提示词。"""
    return _PROMPT_PATH.read_text(encoding="utf-8")


def create_za38_agent(
    model: BaseChatModel | str | None = None,
    tools: list[Any] | None = None,
    middleware: list[Any] | None = None,
    workdir: str | None = None,
) -> Any:
    """创建 za38 编码 agent，启用核心编码工具。

    使用 deepagents 的 FilesystemMiddleware + LocalShellBackend，
    提供与 dcode 相同的核心编码能力：
    - 文件操作：ls, read_file, write_file, edit_file, glob, grep, delete
    - Shell 执行：execute
    - 任务规划：write_todos（TodoListMiddleware 自动添加）

    Args:
        model: ChatModel 实例或模型规格字符串（如 "openai:gpt-4"）。
               如果为 None，使用占位（仅用于测试）。
        tools: 额外工具。核心工具由 middleware 自动注入。
        middleware: 额外中间件。FilesystemMiddleware 会自动添加。
        workdir: 工作目录。默认为当前目录。agent 的文件操作在此目录下执行。

    Returns:
        编译后的 LangGraph agent（CompiledStateGraph），带 astream/ainvoke。
    """
    from za38_agent.providers.za38_gateway import resolve_model as _resolve

    resolved_model = _resolve(model) if model is not None else model

    # LocalShellBackend 提供 filesystem + shell execute 工具
    # 工作目录默认为当前目录，agent 可读写本地文件并执行命令
    backend = LocalShellBackend(cwd=workdir or ".")

    # FilesystemMiddleware 注入文件工具 + execute 工具

    all_middleware = [fs_middleware]
    if middleware:

    # 额外的自定义工具（za38 垂域工具后续在此添加）

 * 解析 za38 venv 中的 Python 可执行文件路径。

export async function resolveVenvPython(): Promise<string | null> {
  const dataDir = getDataDir()
  const venvPath = path.join(dataDir, "venv")
  const pythonPath = path.join(venvPath, "bin", "python")

    return pythonPath
  } catch {
    // Windows 上 python 可能在 Scripts/python.exe

    try {
      await fs.access(winPath)
      return winPath
    } catch {
      return null
    }
  }
}

  return "python3"

import { createElement } from "react"

  // 1. 解析 Python 路径
  const pythonPath = await resolvePython()

  // 2. 启动 Python agent 内核子进程
  const child: ChildProcess = spawn(pythonPath, ["-m", "za38_agent"], {
    stdio: ["pipe", "pipe", "inherit"], // stdin=pipe, stdout=pipe, stderr=inherit
  })

  child.on("exit", (code) => {
    if (code !== 0 && code !== null) {
      console.error(`za38-cli: agent 进程退出，code ${code}`)
    }
    process.exit(code ?? 1)
  })

  // 3. 创建 IPC 客户端
  const stdin = child.stdin!
  const stdout = child.stdout!
  const client = new IpcClient(stdin, stdout)

  // 4. 初始化握手
  try {
    const result = await client.call("initialize", {
      client_info: { name: "za38-cli", version: "0.1.0" },
    }) as any
    console.error(`za38-cli: 已连接到 ${result.server_info.name} v${result.server_info.version}`)
  } catch (err) {
    console.error("za38-cli: 初始化 agent 内核失败:", err)
    child.kill()
    process.exit(1)
  }

  // 5. 启动 opentui 渲染器 + React 应用
  const renderer = await createCliRenderer()
  const root = createRoot(renderer)
  root.render(createElement(App, { client }))

  // 6. 优雅关闭
  const shutdown = async () => {
    try {
      await client.call("shutdown", {})
    } catch {}
    child.kill("SIGTERM")
    setTimeout(() => process.exit(0), 500)
  }

  process.on("SIGINT", shutdown)
  process.on("SIGTERM", shutdown)
  renderer.on("destroy", shutdown)
}

main().catch((err) => {
  console.error("za38-cli: 致命错误:", err)
  process.exit(1)

// packages/cli/scripts/setup/utils.js
const path = require("node:path")
const os = require("node:os")
const fs = require("node:fs/promises")
const { execFile } = require("node:child_process")
const { createWriteStream } = require("node:fs")
const https = require("node:https")

/**
 * 获取 za38 数据目录。
 */
function getDataDir() {
  return path.join(os.homedir(), ".za38")
}

/**
 * 获取当前平台的 uv 下载 URL。
 */
function getUvDownloadUrl(platform, arch) {
  const baseUrl = "https://github.com/astral-sh/uv/releases/latest/download"
  const platformMap = {
    darwin: { arm64: "uv-aarch64-apple-darwin.tar.gz", x64: "uv-x86_64-apple-darwin.tar.gz" },
    linux: { arm64: "uv-aarch64-unknown-linux-gnu.tar.gz", x64: "uv-x86_64-unknown-linux-gnu.tar.gz" },
    win32: { arm64: "uv-aarch64-pc-windows-msvc.zip", x64: "uv-x86_64-pc-windows-msvc.zip" },
  }
  const filename = platformMap[platform]?.[arch]
  if (!filename) throw new Error(`不支持的平台: ${platform}-${arch}`)
  return `${baseUrl}/${filename}`
}

/**
 * 下载当前平台的 uv 二进制。
 * 返回 uv 可执行文件路径。
 */
async function downloadUv(platform, arch) {
  const dataDir = getDataDir()
  const uvDir = path.join(dataDir, "bin")
  await fs.mkdir(uvDir, { recursive: true })

  const ext = platform === "win32" ? ".exe" : ""
  const uvPath = path.join(uvDir, `uv${ext}`)

  // 检查 uv 是否已存在
  try {
    await fs.access(uvPath)
    console.log("za38-cli: uv 已下载")
    return uvPath
  } catch {}

  const url = getUvDownloadUrl(platform, arch)
  console.log(`za38-cli: 从 ${url} 下载 uv...`)

  const isWindows = platform === "win32"
  const archivePath = path.join(uvDir, isWindows ? "uv.zip" : "uv.tar.gz")

  await new Promise((resolve, reject) => {
    const file = createWriteStream(archivePath)
    https.get(url, (response) => {
      if (response.statusCode === 302 || response.statusCode === 301) {
        https.get(response.headers.location, (resp) => {
          resp.pipe(file)
          file.on("finish", () => { file.close(resolve) })
        }).on("error", reject)
      } else {
        response.pipe(file)
        file.on("finish", () => { file.close(resolve) })
      }
    }).on("error", reject)
  })

  // 解压
  if (isWindows) {
    await execFileAsync("tar", ["xf", archivePath, "-C", uvDir])
  } else {
    await execFileAsync("tar", ["xzf", archivePath, "-C", uvDir, "--strip-components=1"])
  }

  if (!isWindows) {
    await fs.chmod(uvPath, 0o755)
  }

  await fs.unlink(archivePath)

  return uvPath
}

/**
 * 执行 uv 命令并返回 stdout。
 */
function execUv(uvPath, args) {
  return execFileAsync(uvPath, args, { stdio: ["ignore", "pipe", "pipe"] })
}

function execFileAsync(cmd, args, options = {}) {
  return new Promise((resolve, reject) => {
    execFile(cmd, args, options, (error, stdout, stderr) => {
      if (error) reject(error)
      else resolve(stdout)
    })
  })
}

module.exports = { getDataDir, getUvDownloadUrl, downloadUv, execUv }
```

    await execUv(uvPath, ["venv", venvPath, "--python", "3.12"])
  } catch (err) {
    console.error("za38-cli: 创建 venv 失败:", err.message)
    throw err
  }

  // 4. 安装 za38-agent
  console.log("za38-cli: 安装 agent 依赖...")
  try {
    const agentPath = path.resolve(__dirname, "../../agent")
    const hasLocalAgent = await fs.access(path.join(agentPath, "pyproject.toml"))
      .then(() => true).catch(() => false)

    if (hasLocalAgent) {
      await execUv(uvPath, ["pip", "install", "--venv", venvPath, "-e", agentPath])
    } else {
      await execUv(uvPath, ["pip", "install", "--venv", venvPath, "za38-agent"])
    }
  } catch (err) {
    console.error("za38-cli: 安装 agent 包失败:", err.message)
    throw err
  }

  console.log("za38-cli: 安装完成!")
  console.log(`za38-cli: venv 创建于 ${venvPath}`)
}

main().catch((err) => {
  console.error("za38-cli: 安装失败:", err.message)
  console.error("za38-cli: 可以尝试运行 'npm rebuild @za38/cli' 重新安装")
  process.exit(1)
})
```
